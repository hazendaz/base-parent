1   package com.hazendaz.rs;

3   import javax.enterprise.context.spi.CreationalContext;
4   import javax.enterprise.inject.spi.Bean;
5   import javax.enterprise.inject.spi.BeanManager;
6   
7   import org.apache.deltaspike.core.api.provider.BeanManagerProvider;
8   import org.glassfish.hk2.utilities.binding.AbstractBinder;
9   
10  public class WebServiceBinder extends AbstractBinder {
11  
12      @Override
13      protected void configure() {
14          this.bind(this.getBean(this.getBeanManager(), HelloWorldService.class)).to(HelloWorldService.class);
15      }
16  
17      @SuppressWarnings("unchecked")
18      private <T> T getBean(final BeanManager beanManager, final Class<T> clazz) {
19          final Bean<T> bean = (Bean<T>) beanManager.getBeans(clazz).iterator().next();
20          final CreationalContext<T> creationalContext = beanManager.createCreationalContext(bean);
21          return (T) beanManager.getReference(bean, clazz, creationalContext);
22      }
23  
24      private BeanManager getBeanManager() {
25          return BeanManagerProvider.getInstance().getBeanManager();
26      }
27  }
