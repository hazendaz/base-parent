1   package com.hazendaz.scheduler;

3   import javax.enterprise.context.RequestScoped;
4   import javax.inject.Inject;
5   
6   import org.apache.deltaspike.cdise.api.ContextControl;
7   import org.apache.deltaspike.core.api.provider.BeanProvider;
8   import org.apache.deltaspike.scheduler.api.Scheduled;
9   import org.quartz.Job;
10  import org.quartz.JobExecutionContext;
11  import org.quartz.JobExecutionException;
12  import org.slf4j.Logger;
13  
14  import com.hazendaz.beans.InventoryItem;
15  
16  @Scheduled(cronExpression = "0/10 * * * * ?")
17  public class DailyCleanup2 implements Job {
18  
19      @Inject
20      private Logger        logger;
21  
22      @Inject
23      private InventoryItem inventoryItem;
24  
25      @Override
26      public void execute(final JobExecutionContext context) throws JobExecutionException {
27  
28          final ContextControl ctxCtrl = BeanProvider.getContextualReference(ContextControl.class);
29  
30          // this will implicitly bind a new RequestContext to the current thread
31          ctxCtrl.startContext(RequestScoped.class);
32  
33          try {
34              // Say Hello to the World and display the date/time
35              this.logger.info("Hello World! - Job 2 {}", Long.valueOf(System.currentTimeMillis()));
36              this.logger.info("logBean info {}", this.inventoryItem);
37          } finally {
38              // stop the RequestContext to ensure that all request-scoped beans get cleaned up.
39              ctxCtrl.stopContext(RequestScoped.class);
40          }
41      }
42  
43  }
