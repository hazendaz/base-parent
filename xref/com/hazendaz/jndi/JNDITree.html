1   package com.hazendaz.jndi;

3   import javax.naming.Context;
4   import javax.naming.InitialContext;
5   import javax.naming.NameClassPair;
6   import javax.naming.NamingEnumeration;
7   import javax.naming.NamingException;
8   
9   public class JNDITree {
10      private Context context     = null;
11  
12      private int     indentLevel = 0;
13  
14      public JNDITree() throws NamingException {
15          this.context = new InitialContext();
16      }
17  
18      public void printJNDITree(final String ct) {
19          try {
20              this.printNE(this.context.list(ct), ct);
21          } catch (final NamingException e) {
22              // Do nothing
23          }
24      }
25  
26      private void decreaseIndent() {
27          this.indentLevel -= 4;
28      }
29  
30      private void increaseIndent() {
31          this.indentLevel += 4;
32      }
33  
34      private void printEntry(final NameClassPair next) {
35          System.out.println(this.printIndent() + "-->" + next);
36      }
37  
38      private String printIndent() {
39          final StringBuilder buf = new StringBuilder(this.indentLevel);
40          for (int i = 0; i < this.indentLevel; i++) {
41              buf.append(" ");
42          }
43          return buf.toString();
44      }
45  
46      private void printNE(final NamingEnumeration<?> ne, final String parentctx) {
47          while (ne.hasMoreElements()) {
48              final NameClassPair next = (NameClassPair) ne.nextElement();
49              this.printEntry(next);
50              this.increaseIndent();
51              this.printJNDITree(parentctx.length() == 0 ? next.getName() : parentctx + "/" + next.getName());
52              this.decreaseIndent();
53          }
54      }
55  }
