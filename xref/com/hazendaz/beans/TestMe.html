20      public static String toImpliedDecimal(final Double value) {
21          return value == null ? "0" : String.format("%.2f", value).replace(".", "");
22      }
23  
24      public static String toMoney(final String value) {
25          if (value != null && value.length() < 3) {
26              return String.format("%.2f", Double.valueOf(Double.valueOf(value).doubleValue() / 100));
27          }
28          return value == null ? "0.00" : String.format("%.2f", Double.valueOf(value));
29      }
