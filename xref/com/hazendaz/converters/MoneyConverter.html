3   import java.text.DecimalFormat;
4   
5   import javax.enterprise.context.ApplicationScoped;
6   import javax.faces.component.UIComponent;
7   import javax.faces.context.FacesContext;
8   import javax.faces.convert.Converter;
9   import javax.faces.convert.FacesConverter;
10  import javax.inject.Named;
11  
12  @ApplicationScoped
13  @Named
14  @FacesConverter(value = "moneyConverter")
15  public class MoneyConverter implements Converter {
16  
17      @Override
18      public Object getAsObject(final FacesContext facesContext, final UIComponent component, final String stringValue) {
19          final DecimalFormat decimalFormat = new DecimalFormat("#.##");
20          return decimalFormat.format(stringValue);
21      }
22  
23      @Override
24      public String getAsString(final FacesContext facesContext, final UIComponent uiComponent, final Object objectValue) {
25          final DecimalFormat decimalFormat = new DecimalFormat("#.##");
26          return decimalFormat.format(objectValue);
27      }
28  
29  }
