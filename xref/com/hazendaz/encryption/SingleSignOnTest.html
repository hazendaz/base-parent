203     private static String _decrypt(final InputStream inputStream, final InputStream keyIn, final char[] passwd)
204             throws Exception {
205 
206         final InputStream in = PGPUtil.getDecoderStream(inputStream);
207         try {
208             final PGPObjectFactory pgpF = new BcPGPObjectFactory(in);
209             PGPEncryptedDataList enc;
210             final Object o = pgpF.nextObject();
211 
212             //
213             // the first object might be a PGP marker packet.
214             //
215             if (o instanceof PGPEncryptedDataList) {
216                 enc = (PGPEncryptedDataList) o;
217             } else {
218                 enc = (PGPEncryptedDataList) pgpF.nextObject();
219             }
220 
221             //
222             // find the secret key
223             //
224             final Iterator<?> it = enc.getEncryptedDataObjects();
225             PGPPrivateKey sKey = null;
226             PGPPublicKeyEncryptedData pbe = null;
227             while (sKey == null && it.hasNext()) {
228                 pbe = (PGPPublicKeyEncryptedData) it.next();
229                 SingleSignOnTest.logger.info("pbe id = {}", Long.valueOf(pbe.getKeyID()));
230                 sKey = SingleSignOnTest.findSecretKey(keyIn, pbe.getKeyID(), passwd);
231             }
232             if (sKey == null) {
233                 in.close();
234                 throw new IllegalArgumentException("secret key for message not found.");
235             }
236             final BcPublicKeyDataDecryptorFactory factory = new BcPublicKeyDataDecryptorFactory(sKey);
237             final InputStream clear = pbe != null ? pbe.getDataStream(factory) : null;
238             final PGPObjectFactory plainFact = new BcPGPObjectFactory(clear);
239             Object message = plainFact.nextObject();
240             if (message instanceof PGPCompressedData) {
241                 final PGPCompressedData cData = (PGPCompressedData) message;
242                 final PGPObjectFactory pgpFact = new BcPGPObjectFactory(cData.getDataStream());
243                 message = pgpFact.nextObject();
244             }
245             final ByteArrayOutputStream baos = new ByteArrayOutputStream();
246             if (message instanceof PGPLiteralData) {
247                 final PGPLiteralData ld = (PGPLiteralData) message;
248                 final InputStream unc = ld.getInputStream();
249                 int ch;
250                 while ((ch = unc.read()) >= 0) {
251                     baos.write(ch);
252                 }
253                 unc.close();
254             } else if (message instanceof PGPOnePassSignatureList) {
255                 if (clear != null) {
256                     clear.close();
257                 }
258                 throw new PGPException("encrypted message contains a signed message - not literal data.");
259             } else {
260                 if (clear != null) {
261                     clear.close();
262                 }
263                 throw new PGPException("message is not a simple encrypted file - type unknown.");
264             }
265             if (pbe != null && pbe.isIntegrityProtected()) {
266                 if (!pbe.verify()) {
267                     SingleSignOnTest.logger.error("message failed integrity check");
268                 } else {
269                     SingleSignOnTest.logger.error("message integrity check passed");
270                 }
271             } else {
272                 SingleSignOnTest.logger.error("no message integrity check");
273             }
274             if (clear != null) {
275                 clear.close();
276             }
277             return baos.toString();
278         } catch (final PGPException e) {
279             System.err.println(e);
280             if (e.getUnderlyingException() != null) {
281                 e.getUnderlyingException().printStackTrace();
282             }
283         }
284         in.close();
285         return null;
286 
287     }

292     private static void _encrypt(final String fileName, final OutputStream outputStream, final PGPPublicKey encKey)
293             throws IOException, PGPException {
294 
295         final OutputStream out = new DataOutputStream(outputStream);
296         final ByteArrayOutputStream bOut = new ByteArrayOutputStream();
297         SingleSignOnTest.logger.info("creating comData...");
298         // get the data from the original file
299         final PGPCompressedDataGenerator comData = new PGPCompressedDataGenerator(CompressionAlgorithmTags.ZIP);
300         PGPUtil.writeFileToLiteralData(comData.open(bOut), PGPLiteralData.BINARY, new File(fileName));
301         comData.close();
302         SingleSignOnTest.logger.info("comData created...");
303         SingleSignOnTest.logger.info("using PGPEncryptedDataGenerator...");
304         // object that encrypts the data
305         final BcPGPDataEncryptorBuilder builder = new BcPGPDataEncryptorBuilder(SymmetricKeyAlgorithmTags.CAST5);
306         final PGPEncryptedDataGenerator cPk = new PGPEncryptedDataGenerator(builder);
307         final BcPublicKeyKeyEncryptionMethodGenerator generator = new BcPublicKeyKeyEncryptionMethodGenerator(encKey);
308         cPk.addMethod(generator);
309         SingleSignOnTest.logger.info("used PGPEncryptedDataGenerator...");
310         // take the outputstream of the original file and turn it into a byte
311         // array
312         final byte[] bytes = bOut.toByteArray();
313         SingleSignOnTest.logger.info("wrote bOut to byte array...");
314         // write the plain text bytes to the armored outputstream
315         final OutputStream cOut = cPk.open(out, bytes.length);
316         cOut.write(bytes);
317         cOut.close();
318         cPk.close();
319         out.close();
320     }

323     // Private class method findSecretKey
324     //
325     private static PGPPrivateKey findSecretKey(final InputStream keyIn, final long keyID, final char[] pass)
326             throws IOException, PGPException {
327 
328         final PGPSecretKeyRingCollection pgpSec = new BcPGPSecretKeyRingCollection(PGPUtil.getDecoderStream(keyIn));
329         final PGPSecretKey pgpSecKey = pgpSec.getSecretKey(keyID);
330         if (pgpSecKey == null) {
331             return null;
332         }
333         return pgpSecKey.extractPrivateKey(new BcPBESecretKeyDecryptorBuilder(new BcPGPDigestCalculatorProvider())
334                 .build(pass));
335 
336     }

341     public byte[] readFile(final File file) throws IOException {
342 
343         final BufferedInputStream fis = new BufferedInputStream(new FileInputStream(file));
344         final byte[] buf = new byte[8192];
345         int numRead = 0;
346         final ByteArrayOutputStream baos = new ByteArrayOutputStream();
347         while ((numRead = fis.read(buf)) > 0) {
348             baos.write(buf, 0, numRead);
349         }
350         fis.close();
351         final byte[] returnVal = baos.toByteArray();
352         baos.close();
353         return returnVal;
