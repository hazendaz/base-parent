44      public final static PGPKeyRingGenerator generateKeyRingGenerator(final String id, final char[] pass,
45              final int s2kcount) throws Exception {
46          // This object generates individual key-pairs.
47          final RSAKeyPairGenerator kpg = new RSAKeyPairGenerator();
48  
49          // Boilerplate RSA parameters, no need to change anything
50          // except for the RSA key-size (2048). You can use whatever
51          // key-size makes sense for you -- 4096, etc.
52          kpg.init(new RSAKeyGenerationParameters(BigInteger.valueOf(0x10001), new SecureRandom(), 2048, 12));
53  
54          // First create the master (signing) key with the generator.
55          final PGPKeyPair rsakp_sign = new BcPGPKeyPair(PublicKeyAlgorithmTags.RSA_SIGN, kpg.generateKeyPair(),
56                  new Date());
57          // Then an encryption subkey.
58          final PGPKeyPair rsakp_enc = new BcPGPKeyPair(PublicKeyAlgorithmTags.RSA_ENCRYPT, kpg.generateKeyPair(),
59                  new Date());
60  
61          // Add a self-signature on the id
62          final PGPSignatureSubpacketGenerator signhashgen = new PGPSignatureSubpacketGenerator();
63  
64          // Add signed metadata on the signature.
65          // 1) Declare its purpose
66          signhashgen.setKeyFlags(false, KeyFlags.SIGN_DATA | KeyFlags.CERTIFY_OTHER);
67          // 2) Set preferences for secondary crypto algorithms to use
68          // when sending messages to this key.
69          signhashgen.setPreferredSymmetricAlgorithms(false, new int[] { SymmetricKeyAlgorithmTags.AES_256,
70                  SymmetricKeyAlgorithmTags.AES_192, SymmetricKeyAlgorithmTags.AES_128 });
71          signhashgen.setPreferredHashAlgorithms(false, new int[] { HashAlgorithmTags.SHA256, HashAlgorithmTags.SHA1,
72                  HashAlgorithmTags.SHA384, HashAlgorithmTags.SHA512, HashAlgorithmTags.SHA224, });
73          // 3) Request senders add additional checksums to the
74          // message (useful when verifying unsigned messages.)
75          signhashgen.setFeature(false, Features.FEATURE_MODIFICATION_DETECTION);
76  
77          // Create a signature on the encryption subkey.
78          final PGPSignatureSubpacketGenerator enchashgen = new PGPSignatureSubpacketGenerator();
79          // Add metadata to declare its purpose
80          enchashgen.setKeyFlags(false, KeyFlags.ENCRYPT_COMMS | KeyFlags.ENCRYPT_STORAGE);
81  
82          // Objects used to encrypt the secret key.
83          final PGPDigestCalculator sha1Calc = new BcPGPDigestCalculatorProvider().get(HashAlgorithmTags.SHA1);
84          final PGPDigestCalculator sha256Calc = new BcPGPDigestCalculatorProvider().get(HashAlgorithmTags.SHA256);
85  
86          // bcpg 1.48 exposes this API that includes s2kcount. Earlier
87          // versions use a default of 0x60.
88          final PBESecretKeyEncryptor pske = new BcPBESecretKeyEncryptorBuilder(SymmetricKeyAlgorithmTags.AES_256,
89                  sha256Calc, s2kcount).build(pass);
90  
91          // Finally, create the keyring itself. The constructor
92          // takes parameters that allow it to generate the self
93          // signature.
94          final PGPKeyRingGenerator keyRingGen = new PGPKeyRingGenerator(PGPSignature.POSITIVE_CERTIFICATION, rsakp_sign,
95                  id, sha1Calc, signhashgen.generate(), null, new BcPGPContentSignerBuilder(rsakp_sign.getPublicKey()
96                          .getAlgorithm(), HashAlgorithmTags.SHA1), pske);
97  
98          // Add our encryption subkey, together with its signature.
99          keyRingGen.addSubKey(rsakp_enc, enchashgen.generate(), null);
100         return keyRingGen;
101     }

116     public PGPPublicKeyRing createPublicKey(final String email, final char[] password) {
117 
118         // Checks for max allowed value --- without fix in Java, it should say
119         // 128.
120         try {
121             final int maxKeyLen = Cipher.getMaxAllowedKeyLength("AES");
122             this.logger.info("Allowed Key Length is: {}", String.valueOf(maxKeyLen));
123         } catch (final NoSuchAlgorithmException e1) {
124             this.logger.error(e1.toString());
125         }
126 
127         KeyPair keyPair = null;
128         KeyPair keyPair2 = null;
129         PGPPublicKeyRing pkr = null;
130         try {
131 
132             final KeyPairGenerator keyPairGen = KeyPairGenerator.getInstance("RSA", "BC");
133             keyPairGen.initialize(4096);
134 
135             keyPair = keyPairGen.generateKeyPair();
136             keyPair2 = keyPairGen.generateKeyPair();
137 
138             final AsymmetricKeyParameter privKey = PrivateKeyFactory.createKey(keyPair.getPrivate().getEncoded());
139             final AsymmetricKeyParameter pubKey = PublicKeyFactory.createKey(keyPair2.getPublic().getEncoded());
140             final AsymmetricCipherKeyPair privKeyParams = new AsymmetricCipherKeyPair(pubKey, privKey);
141 
142             final PGPSignatureSubpacketGenerator hashedGen = new PGPSignatureSubpacketGenerator();
143 
144             hashedGen.setKeyFlags(true, KeyFlags.CERTIFY_OTHER | KeyFlags.SIGN_DATA | KeyFlags.ENCRYPT_COMMS
145                     | KeyFlags.ENCRYPT_STORAGE);
146 
147             hashedGen.setPreferredCompressionAlgorithms(false, new int[] { CompressionAlgorithmTags.ZIP });
148 
149             hashedGen.setPreferredHashAlgorithms(false, new int[] { HashAlgorithmTags.SHA1 });
150 
151             hashedGen.setPreferredSymmetricAlgorithms(false, new int[] { SymmetricKeyAlgorithmTags.AES_128 });
152 
153             final PGPKeyPair secretKey2 = new BcPGPKeyPair(PublicKeyAlgorithmTags.RSA_GENERAL, privKeyParams,
154                     new Date());
155 
156             final PGPKeyRingGenerator keyRingGen = TestKeyGen.generateKeyRingGenerator(email + "<" + email
157                     + "@homail.com>", password, 0xc0);
158 
159             keyRingGen.addSubKey(secretKey2);
160 
161             pkr = keyRingGen.generatePublicKeyRing();
162 
163         } catch (final PGPException e) {
164             this.logger.error(e.toString());
165             this.logger.error(e.getUnderlyingException().toString());
166         } catch (final Exception e) {
167             this.logger.error(e.toString());
168         }
169         return pkr;
170

173     // Note: s2kcount is a number between 0 and 0xff that controls the
174     // number of times to iterate the password hash before use. More
175     // iterations are useful against offline attacks, as it takes more
176     // time to check each password. The actual number of iterations is
177     // rather complex, and also depends on the hash function in use.
178     // Refer to Section 3.7.1.3 in rfc4880.txt. Bigger numbers give
179     // you more iterations. As a rough rule of thumb, when using
180     // SHA256 as the hashing function, 0x10 gives you about 64
181     // iterations, 0x20 about 128, 0x30 about 256 and so on till 0xf0,
182     // or about 1 million iterations. The maximum you can go to is
183     // 0xff, or about 2 million iterations. I'll use 0xc0 as a
184     // default -- about 130,000 iterations.
185
