3   import java.io.IOException;
4   import java.util.ArrayList;
5   import java.util.List;
6   import java.util.Scanner;
7   
8   import javax.faces.application.FacesMessage;
9   import javax.faces.component.UIComponent;
10  import javax.faces.context.FacesContext;
11  import javax.faces.validator.ValidatorException;
12  import javax.faces.view.ViewScoped;
13  import javax.inject.Named;
14  import javax.servlet.http.Part;
15  
16  import lombok.Data;
17  
18  @Data
19  @Named
20  @ViewScoped
21  public class Bean {
22      private Part   file;
23      private String fileContent;
24  
25      /**
26       * Upload bean using standard jsf
27       */
28      public void upload() {
29          try {
30              this.fileContent = new Scanner(this.file.getInputStream()).useDelimiter("\\A").next();
31          } catch (final IOException e) {
32              final FacesMessage msg = new FacesMessage(FacesMessage.SEVERITY_ERROR, "error uploading file", null);
33              FacesContext.getCurrentInstance().addMessage(null, msg);
34          }
35      }
36  
37      /**
38       * Validate file upload using standard jsf
39       * 
40       * @param ctx
41       *            faces context
42       * @param comp
43       *            ui component
44       * @param value
45       *            object
46       */
47      public void validateFile(final FacesContext ctx, final UIComponent comp, final Object value) {
48          final List<FacesMessage> msgs = new ArrayList<FacesMessage>();
49          final Part part = (Part) value;
50          if (part.getSize() > 1024) {
51              msgs.add(new FacesMessage("file too big"));
52          }
53          if (!"text/plain".equals(part.getContentType())) {
54              msgs.add(new FacesMessage("not a text file"));
55          }
56          if (!msgs.isEmpty()) {
57              throw new ValidatorException(msgs);
58          }
59      }
60  }
