3   import java.util.ArrayList;
4   import java.util.List;
5   
6   import javax.enterprise.event.Observes;
7   import javax.enterprise.inject.spi.AfterDeploymentValidation;
8   import javax.enterprise.inject.spi.BeanManager;
9   import javax.enterprise.inject.spi.Extension;
10  import javax.enterprise.inject.spi.InjectionTarget;
11  import javax.enterprise.inject.spi.ProcessInjectionTarget;
12  
13  public class AfterDeployExtension implements Extension {
14  
15      private final List<InjectionTarget<?>> methods = new ArrayList<InjectionTarget<?>>();
16  
17      public <T> void collect(@Observes final ProcessInjectionTarget<T> event) {
18          if (event.getAnnotatedType().isAnnotationPresent(AfterDeploy.class)) {
19              this.methods.add(event.getInjectionTarget());
20          }
21      }
22  
23      /**
24       * @param event
25       *            afterDeploymentEvent
26       * @param beanManager
27       *            cdi bean manager
28       */
29      public void load(@Observes final AfterDeploymentValidation event, final BeanManager beanManager) {
30          beanManager.getBeans("applicationTest");
31          for (final InjectionTarget<?> bean : this.methods) {
32              bean.postConstruct(null);
33          }
34      }
35  
36  }
