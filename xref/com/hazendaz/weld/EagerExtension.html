1   package com.hazendaz.weld;
2   
3   import java.util.ArrayList;
4   import java.util.List;
5   
6   import javax.enterprise.context.ApplicationScoped;
7   import javax.enterprise.event.Observes;
8   import javax.enterprise.inject.spi.AfterDeploymentValidation;
9   import javax.enterprise.inject.spi.Bean;
10  import javax.enterprise.inject.spi.BeanManager;
11  import javax.enterprise.inject.spi.Extension;
12  import javax.enterprise.inject.spi.ProcessBean;
13  
14  import org.slf4j.Logger;
15  import org.slf4j.LoggerFactory;
16  
17  public class EagerExtension implements Extension {
18      private final Logger        logger                = LoggerFactory.getLogger(EagerExtension.class);
19      private final List<Bean<?>> eagerBeansList        = new ArrayList<Bean<?>>();
20      private final List<Bean<?>> invalidEagerBeansList = new ArrayList<Bean<?>>();
21  
22      public <T> void collect(@Observes final ProcessBean<T> event) {
23          if (event.getAnnotated().isAnnotationPresent(Eager.class)) {
24              if (event.getAnnotated().isAnnotationPresent(ApplicationScoped.class)) {
25                  this.eagerBeansList.add(event.getBean());
26              } else {
27                  this.invalidEagerBeansList.add(event.getBean());
28              }
29          }
30      }
31  
32      /**
33       * @param event
34       *            afterDeploymentValidation
35       * @param beanManager
36       *            cdi bean manager
37       */
38      public void load(@Observes final AfterDeploymentValidation event, final BeanManager beanManager) {
39          for (final Bean<?> bean : this.eagerBeansList) {
40              // note: toString() is important to instantiate the bean
41              beanManager.getReference(bean, bean.getBeanClass(), beanManager.createCreationalContext(bean)).toString();
42          }
43          for (final Bean<?> bean : this.invalidEagerBeansList) {
44              this.logger.error("@Eager annotation not used with @ApplicationScoped in {}",
45                      beanManager.getReference(bean, bean.getBeanClass(), beanManager.createCreationalContext(bean))
46                              .toString());
47          }
48      }
49  }
